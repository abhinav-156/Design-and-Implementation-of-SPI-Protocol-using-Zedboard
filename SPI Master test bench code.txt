`timescale 1ns/1ps
module tb_spi;
    reg clk=0, reset_n=0, start=0;
    wire MOSI,SCLK,SS;
    wire done;

    spi_master uut(
        .clk(clk), .reset_n(reset_n),
        .start(start),
        .mosi_data(8'hAB),
        .done(done),
        .SCLK(SCLK),
        .MOSI(MOSI),
        .SS(SS)
    );

    always #5 clk=~clk;

    initial begin
        #20 reset_n=1;
        #30 start=1;
        #10 start=0;
        #1000 $finish;
    end
endmodule
